22           public static ArrayList<String> readWordsFromCSV(String file) throws FileNotFoundException {

24               sc.useDelimiter(" |,"); 
25               ArrayList<String> words=new ArrayList<>(); 
26               String str; 
27               while(sc.hasNext()){ 
28                   str=sc.next(); 
29                   str=str.trim(); 
30                   str=str.toLowerCase(); 
31                   words.add(str); 
32                   //Collections.sort(words); 
33               } 
34               return words; 
35           }
